def get_fii_proxy(self):
    """
    Derives a FII buy/sell signal from large-cap volume vs 20-day avg.
    High volume + price up = likely institutional buying.
    High volume + price down = likely institutional selling.
    This is an approximation only. Not real FII data.
    """
    proxy_tickers = ['HDFCBANK.NS', 'RELIANCE.NS', 'INFY.NS', 'TCS.NS', 'ICICIBANK.NS']
    try:
        data = yf.download(proxy_tickers, period='25d', progress=False)
        volume = data['Volume']
        close = data['Close']
        avg_vol = volume.iloc[:-5].mean()
        recent_vol = volume.iloc[-5:].mean()
        recent_change = (close.iloc[-1] - close.iloc[-5]) / close.iloc[-5] * 100
        vol_ratio = recent_vol / avg_vol
        avg_change = recent_change.mean()
        if vol_ratio.mean() > 1.15 and avg_change > 0.3:
            return {'signal': 'BUYING', 'confidence': 'MODERATE', 'note': 'Volume proxy'}
        elif vol_ratio.mean() > 1.15 and avg_change < -0.3:
            return {'signal': 'SELLING', 'confidence': 'MODERATE', 'note': 'Volume proxy'}
        return {'signal': 'NEUTRAL', 'confidence': 'LOW', 'note': 'Volume proxy'}
    except Exception as e:
        logging.error(f'FII proxy error: {e}')
        return {'signal': 'N/A', 'confidence': 'NONE', 'note': 'Data unavailable'}

@app.get('/api/commodities')
async def get_commodities():
    """Live India commodity prices with signal recommendations."""
    tickers = {
        'GOLD': 'GC=F',       # Gold futures (USD, convert to INR)
        'SILVER': 'SI=F',     # Silver futures
        'COPPER': 'HG=F',     # Copper futures
        'BRENT': 'BZ=F',      # Brent crude
        'NATGAS': 'NG=F',     # Natural gas
    }
    try:
        data = yf.download(list(tickers.values()), period='30d', progress=False)
        usd_inr = yf.Ticker('USDINR=X').fast_info['lastPrice']
        result = {}
        for name, ticker in tickers.items():
            closes = data['Close'][ticker].dropna()
            if len(closes) < 2:
                continue
            price_usd = closes.iloc[-1]
            price_inr = price_usd * usd_inr
            change_1d = (closes.iloc[-1] - closes.iloc[-2]) / closes.iloc[-2] * 100
            change_30d = (closes.iloc[-1] - closes.iloc[0]) / closes.iloc[0] * 100
            # Simple RSI (14-period approximation)
            delta = closes.diff().dropna()
            gain = delta.clip(lower=0).rolling(14).mean().iloc[-1]
            loss = (-delta.clip(upper=0)).rolling(14).mean().iloc[-1]
            rsi = 100 - (100 / (1 + gain/loss)) if loss != 0 else 50
            # Signal logic
            if rsi < 35 and change_30d < -5:
                signal = 'BUY'; confidence = 78
            elif rsi > 65 and change_30d > 5:
                signal = 'SELL'; confidence = 72
            else:
                signal = 'HOLD'; confidence = 55
            result[name] = {
                'price_inr': round(price_inr, 2),
                'price_usd': round(price_usd, 2),
                'change_1d': round(change_1d, 2),
                'change_30d': round(change_30d, 2),
                'rsi': round(rsi, 1),
                'signal': signal,
                'confidence': confidence,
            }
        return result
    except Exception as e:
        return {'error': str(e)}

@app.get('/api/risk')
async def get_risk():
    """Global risk score and event shock feed."""
    gdelt = cache_get('gdelt_india')
    events = gdelt.get('events', []) if gdelt else []
    gti = gdelt.get('gti', 50) if gdelt else 50
    
    # Compute shock scores from GDELT tone
    event_feed = []
    for evt in events[:8]:
        tone = evt.get('tone', 0)
        shock = min(99, int(abs(tone) * 10 + 20))
        category = categorize_event(evt['title'])  # see below
        event_feed.append({
            'title': evt['title'],
            'domain': evt['domain'],
            'shock': shock,
            'category': category,
            'instrument': map_event_to_instrument(evt),
        })
    
    # Risk score contributing factors
    factors = {
        'event_volume_24h': min(100, len(events) * 4),
        'avg_severity': int(abs(sum(e.get('tone',0) for e in events) / max(len(events),1)) * 10),
        'gti': gti,
    }
    
    return {
        'risk_score': int(gti),
        'risk_label': 'RISK OFF' if gti > 60 else 'NEUTRAL' if gti > 35 else 'RISK ON',
        'factors': factors,
        'event_feed': event_feed,
    }

def categorize_event(title):
    t = title.lower()
    if any(w in t for w in ['oil', 'energy', 'opec', 'crude']): return 'ENERGY'
    if any(w in t for w in ['rate', 'fed', 'rbi', 'inflation']): return 'RATES'
    if any(w in t for w in ['trade', 'tariff', 'export']): return 'TRADE'
    if any(w in t for w in ['war', 'military', 'conflict', 'attack']): return 'GEOPOLITICAL'
    return 'MACRO'

def map_event_to_instrument(evt):
    title = evt.get('title', '').lower()
    country = evt.get('country', '')
    if 'oil' in title or 'crude' in title or 'opec' in title: return 'BRENT'
    if 'rupee' in title or 'inr' in title or country in ['IN', 'IND']: return 'USD/INR'
    if 'fed' in title or 'rate' in title: return 'NIFTY 50'
    if 'gold' in title: return 'GOLD'
    return 'NIFTY 50'

df = pd.read_csv('../data/training_data.csv', index_col=0)

FEATURES = ['nifty_change', 'oil_change', 'gold_change', 'vix_change',
            'usd_inr', 'nifty_5d_mom', 'nifty_20d_mom', 'oil_20d_mom', 'vix_level']

X = df[FEATURES].values
y = df['target'].values

# Time-series split — NEVER use random split on time series data
# This is a common mistake that inflates accuracy artificially
tscv = TimeSeriesSplit(n_splits=5)

params = {
    'objective': 'multiclass',
    'num_class': 3,
    'metric': 'multi_logloss',
    'verbosity': -1,
    'boosting_type': 'gbdt',
    'num_leaves': 31,
    'learning_rate': 0.05,
    'feature_fraction': 0.9,
    'min_child_samples': 20,
}

# Train on last fold (most recent data) for production model
for fold, (train_idx, test_idx) in enumerate(tscv.split(X)):
    X_train, X_test = X[train_idx], X[test_idx]
    y_train, y_test = y[train_idx], y[test_idx]
    train_data = lgb.Dataset(X_train, label=y_train)
    model = lgb.train(params, train_data, num_boost_round=200)
    preds = np.argmax(model.predict(X_test), axis=1)
    print(f'Fold {fold+1}:')
    print(classification_report(y_test, preds, target_names=['NEUTRAL','BULLISH','BEARISH']))

Part A — Backtesting Module
The backtester evaluates how accurate your signal engine has been historically. It compares each generated signal against what the market actually did the next day.
Step 1 — Create src/backtester.py
import yfinance as yf
import pandas as pd
import numpy as np
from datetime import datetime, timedelta

class Backtester:
    """
    Runs the signal engine against historical data and
    computes accuracy, win rate, and risk-reward statistics.
    """
    def __init__(self, signal_engine):
        self.engine = signal_engine

    def run(self, start_date='2023-01-01', end_date=None):
        if end_date is None:
            end_date = datetime.today().strftime('%Y-%m-%d')
        
        tickers = ['^NSEI', 'BZ=F', 'GC=F', '^INDIAVIX', 'USDINR=X']
        data = yf.download(tickers, start=start_date, end=end_date)
        closes = data['Close']
        closes.columns = ['nifty', 'oil', 'gold', 'vix', 'usdinr']
        closes.dropna(inplace=True)
        
        results = []
        for i in range(20, len(closes) - 1):
            row = closes.iloc[i]
            prev = closes.iloc[i-1]
            features = {
                'nifty_change': (row.nifty - prev.nifty) / prev.nifty * 100,
                'oil_change': (row.oil - prev.oil) / prev.oil * 100,
                'gold_change': (row.gold - prev.gold) / prev.gold * 100,
                'vix_change': (row.vix - prev.vix) / prev.vix * 100,
                'usd_inr': row.usdinr,
                'nifty_5d_mom': (row.nifty - closes.iloc[i-5].nifty) / closes.iloc[i-5].nifty * 100,
                'nifty_20d_mom': (row.nifty - closes.iloc[i-20].nifty) / closes.iloc[i-20].nifty * 100,
                'oil_20d_mom': (row.oil - closes.iloc[i-20].oil) / closes.iloc[i-20].oil * 100,
                'vix_level': row.vix,
                'bullish_count': 0, 'bearish_count': 0,
                'geo_high_count': 0, 'avg_severity': 0,
                'oil_risk_count': 0, 'rbi_policy_count': 0,
                'fii_proxy': 'neutral',
            }
            from indices import compute_composite
            indices = compute_composite(features)
            signal = self.engine.generate_signal(features, indices)
            
            next_return = (closes.iloc[i+1].nifty - row.nifty) / row.nifty * 100
            actual = 'BULLISH' if next_return > 0.3 else 'BEARISH' if next_return < -0.3 else 'NEUTRAL'
            
            results.append({
                'date': closes.index[i],
                'signal': signal['direction'],
                'confidence': signal['confidence'],
                'actual': actual,
                'next_return': next_return,
                'correct': signal['direction'] == actual,
            })
        
        df = pd.DataFrame(results)
        directional = df[df['signal'] != 'NEUTRAL']
        print(f'Total signals: {len(df)}')
        print(f'Directional signals: {len(directional)}')
        print(f'Win rate: {directional["correct"].mean()*100:.1f}%')
        print(f'Average return on signal days: {directional["next_return"].mean():.2f}%')
        return df

Honest Caveats and Known Limitations
This is a student project and it should be represented accurately. Here are the known limitations that should be disclosed in your README and any presentation.
Data Limitations
FII/DII data is a volume-derived proxy, not real institutional flow data. Real FII data requires NSDL subscription or Upstox API integration.
GDELT processes news articles, not primary intelligence sources. Its country_scores can be noisy and lag real-world events by hours.
yfinance data has a 15-minute delay for NSE instruments during market hours. It is not tick-level or even 1-minute data.
The model is trained on index-level price data only. It does not account for earnings surprises, policy announcements not in news, or microstructure effects.
Model Limitations
A win rate above 55% on directional signals is good for this type of model. Do not expect 70%+ accuracy. Markets are hard to predict.
The model is trained on 2021-2024 data which includes COVID recovery, rate hike cycle, and geopolitical shocks. Future market regimes may differ.
FinBERT fine-tuning on 500 headlines is minimal. A production NLP system would use 10,000+ labeled examples.

What This Project Is Not
⚠  Not a trading bot. Signals are informational only and should never be acted upon without independent analysis.

⚠  Not production-grade financial infrastructure. It lacks the redundancy, latency guarantees, and regulatory compliance required for real trading systems.

def fetch_options_pcr() → dict:
  session = nse_session_manager.get_session()
  resp = session.get(
    'https://www.nseindia.com/api/option-chain-indices?symbol=NIFTY',
    timeout=10
  )
  data = resp.json()
  records = data['records']['data']

  total_put_oi = sum(r['PE']['openInterest'] for r in records if 'PE' in r)
  total_call_oi = sum(r['CE']['openInterest'] for r in records if 'CE' in r)

  if total_call_oi == 0: return {}
  pcr = round(total_put_oi / total_call_oi, 3)

  interpretation:
    pcr > 1.3 → BULLISH (put heavy = market expects support)
    pcr < 0.7 → BEARISH (call heavy = market expects resistance)
    else → NEUTRAL

  return {
    'pcr': pcr,
    'signal': interpretation,
    'total_put_oi': total_put_oi,
    'total_call_oi': total_call_oi,
    'timestamp': datetime.now(IST).isoformat()
  }

  return await loop.run_in_executor(None, fetcher.fetch_options_pcr)

WHAT BREAKS FIRST
NSE API blocks sessions — yfinance fallback handles this. You will see 'yfinance Fallback' badge. Not a bug.
HuggingFace cold start — first FinBERT call takes 20-30s after inactivity. All subsequent calls are fast. Normal.
GDELT empty response — cached() decorator serves last good response. Do not remove the cache.
Render sleep — free tier sleeps after 15min idle. First post-sleep request = ~30s. Normal.
NSE session cookie expires — reset() method in NSESessionManager handles re-warmup automatically.
LightGBM training fails on thin data — train_model.py skips training if < 30 rows. Falls back to unavailable state.
RBI page structure changes — repo rate scraping breaks if RBI redesigns their press release page. Use fallback hardcoded value.

WHAT NOT TO DO
Do not add any paid API — Bloomberg, Quandl, Polygon, Alpha Vantage. Free stack is sufficient.
Do not use localStorage or sessionStorage in React. Not supported in this environment.
Do not call GDELT more than once per 15 minutes. Rate limit is real.
Do not put secrets in render.yaml. Use Render dashboard environment variables.
Do not skip phases. Each phase depends on the previous one being live and verified.
Do not show Entry/TP/SL prices on signal cards. This is intelligence, not a trading advisory tool.
Do not use random train/test split on time series. Always use TimeSeriesSplit.
Do not shuffle time series data before training. Past leaks into future — results will be fake.
